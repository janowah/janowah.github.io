#include <iostream>

/* iterative */
int factorial ( int number )
{
  if (number == 1)
    return 1;
  else
    return number * factorial (number -1);
}

main()
{
  int AnyNumber;

  std::cin >> AnyNumber;

  std::cout << "The factorial of " << AnyNumber << " is " << factorial (AnyNumber) << std::endl;
}
