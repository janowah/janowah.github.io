Fibonacci

#include <iostream>

int Fibonacci ( int n )
{
  if ( n == 0  || n == 1)
    return n;
  else
    return Fibonacci ( n-1 ) + Fibonacci (n-2);

}

main()
{
  // int AnyNumber;
  //  std::cin >> AnyNumber;

  //  std::cout << "Fib " << AnyNumber << " is " << Fibonacci (AnyNumber) << std::endl;

  for ( int i = 0 ; i < 9 ; i++)
    {
      std::cout << Fibonacci (i) << std::endl ;
    }
}
