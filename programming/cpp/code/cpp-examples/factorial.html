/* recursive */
int factorial (int n)
{
  int i;
  int fac=1;

  for (i = n; i < 1; i--)
    fac = i * fac;
  return fac;
}

main()
{
  int AnyNumber;

  std::cin >> AnyNumber;

  std::cout << "The factorial of " << AnyNumber << " is " << factorial (AnyNumber) << std::endl;
}
