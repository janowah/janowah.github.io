 2: import java.io.IOException;

15: public class Anagrams2 {
16: 
17:   public static void main(String[] args) {
18:     int minGroupSize = Integer.parseInt(args[1]);
19: 
20:     // Read words from file and put into simulated multimap
21:     Map<String, List<String>> m
22:         = new HashMap<String, List<String>>();
23:     try {
24:       Scanner s = new Scanner(new File(args[0]));
25:       while (s.hasNext()) {
26:         String word = s.next();
27:         String alpha = alphabetize(word);
28:         List<String> l = m.get(alpha);
29:         if (l == null) {
30:           m.put(alpha, l = new ArrayList<String>());
31:         }
32:         l.add(word);
33:       }
34:     } catch (IOException e) {
35:       System.err.println(e);
36:       System.exit(1);
37:     }
38: 
39:     // Make a List of all permutation groups above size threshold
40:     List<List<String>> winners = new ArrayList<List<String>>();
41:     for (List<String> l : m.values()) {
42:       if (l.size() >= minGroupSize) {
43:         winners.add(l);
44:       }
45:     }
46: 
47:     // Sort permutation groups according to size
48:     Collections.sort(winners, new Comparator<List<String>>() {
49:         public int compare(List<String> o1, List<String> o2) {
50:           return o2.size() - o1.size();
51:         }
52:       });
53: 
54:     // Print permutation groups
55:     for (List<String> l : winners) {
56:       System.out.println(l.size() + ": " + l);
57:     }
58:   }
59: 
60:   private static String alphabetize(String s) {
61:     char[] a = s.toCharArray();
62:     Arrays.sort(a);
63:     return new String(a);
64:   }
