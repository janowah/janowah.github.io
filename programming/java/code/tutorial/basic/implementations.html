 9: public class FileList {
10:   public static void main(String[] args) {
11:     final int assumedLineLength = 50;
12:     File file = new File(args[0]);
13:     List<String> fileList = new ArrayList<String>((int) (file.length() / assumedLineLength) * 2);
14:     BufferedReader reader = null;

16:     try {
17:       reader = new BufferedReader(new FileReader(file));
18:       for (String line = reader.readLine(); line != null; line = reader.readLine()) {
19:         fileList.add(line);
20:         lineCount++;
21:       }
22:     } catch (IOException e) {
23:       System.err.format("Could not read %s: %s%n", file, e);
24:       System.exit(1);
25:     } finally {
26:       if (reader != null) {
27:         try {
28:           reader.close();
29:         } catch (IOException e) {
30:           ;
31:         }
32:       }
33:     }
34:     int repeats = Integer.parseInt(args[1]);
35:     Random random = new Random();
36:     for (int i = 0; i < repeats; i++) {
37:       System.out.format("%d: %s%n", i, fileList.get(random.nextInt(lineCount - 1)));
38:     }
39:   }
