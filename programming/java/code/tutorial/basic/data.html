data/Anagram.java

 1: public class Anagram {
 2: 
 3:   /**
 4:    * Tests whether the passed-in strings are anagrams -- containing the exact same number of each
 5:    * letter. Punctuation, case, and order don't matter.
 6:    *
 7:    * @return true if the strings are anagrams; otherwise, false
 8:    */
 9:   public static boolean areAnagrams(String string1, String string2) {
10: 
11:     String workingCopy1 = removeJunk(string1);
12:     String workingCopy2 = removeJunk(string2);
13: 
14:     workingCopy1 = workingCopy1.toLowerCase();
15:     workingCopy2 = workingCopy2.toLowerCase();
16: 
17:     workingCopy1 = sort(workingCopy1);
18:     workingCopy2 = sort(workingCopy2);
19: 
20:     return workingCopy1.equals(workingCopy2);
21:   }
22: 
23:   /**
24:    * Removes punctuation & spaces -- everything except letters from the passed-in string.
25:    *
26:    * @return a stripped copy of the passed-in string
27:    */
28:   protected static String removeJunk(String string) {
29:     int i;
30:     int len = string.length();
31:     StringBuilder dest = new StringBuilder(len);
32:     char c;
33: 
34:     for (i = (len - 1); i >= 0; i--) {
35:       c = string.charAt(i);
36:       if (Character.isLetter(c)) {
37:         dest.append(c);
38:       }
39:     }
40: 
41:     return dest.toString();
42:   }
43: 
44:   /**
45:    * Sorts the passed-in string.
46:    *
47:    * @return a sorted copy of the passed-in string
48:    */
49:   protected static String sort(String string) {
50:     char[] charArray = string.toCharArray();
51: 
52:     java.util.Arrays.sort(charArray);
53: 
54:     return new String(charArray);
55:   }
56: 
57:   public static void main(String[] args) {
58:     String string1 = "Cosmo and Laine:";
59:     String string2 = "Maid, clean soon!";
60: 
61:     System.out.println();
62:     System.out.println("Testing whether the following " + "strings are anagrams:");
63:     System.out.println("    String 1: " + string1);
64:     System.out.println("    String 2: " + string2);
65:     System.out.println();
66: 
67:     if (areAnagrams(string1, string2)) {
68:       System.out.println("They ARE anagrams!");
69:     } else {
70:       System.out.println("They are NOT anagrams!");
71:     }
72:     System.out.println();
73:   }
74: }

data/Filename.java

 1: public class Filename {
 2:   private String fullPath;
 3:   private char pathSeparator;
 4:   private char extensionSeparator;
 5: 
 6:   public Filename(String str, char sep, char ext) {
 7:     fullPath = str;
 8:     pathSeparator = sep;
 9:     extensionSeparator = ext;
10:   }
11: 
12:   public String extension() {
13:     int dot = fullPath.lastIndexOf(extensionSeparator);
14:     return fullPath.substring(dot + 1);
15:   }
16: 
17:   // gets filename without extension
18:   public String filename() {
19:     int dot = fullPath.lastIndexOf(extensionSeparator);
20:     int sep = fullPath.lastIndexOf(pathSeparator);
21:     return fullPath.substring(sep + 1, dot);
22:   }
23: 
24:   public String path() {
25:     int sep = fullPath.lastIndexOf(pathSeparator);
26:     return fullPath.substring(0, sep);
27:   }
28: }

data/Palindrome.java

 1: public class Palindrome {
 2: 
 3:   public static boolean isPalindrome(String stringToTest) {
 4:     String workingCopy = removeJunk(stringToTest);
 5:     String reversedCopy = reverse(workingCopy);
 6: 
 7:     return reversedCopy.equalsIgnoreCase(workingCopy);
 8:   }
 9: 
10:   protected static String removeJunk(String string) {
11:     int i, len = string.length();
12:     StringBuilder dest = new StringBuilder(len);
13:     char c;
14: 
15:     for (i = (len - 1); i >= 0; i--) {
16:       c = string.charAt(i);
17:       if (Character.isLetterOrDigit(c)) {
18:         dest.append(c);
19:       }
20:     }
21: 
22:     return dest.toString();
23:   }
24: 
25:   protected static String reverse(String string) {
26:     StringBuilder sb = new StringBuilder(string);
27: 
28:     return sb.reverse().toString();
29:   }
30: 
31:   public static void main(String[] args) {
32:     String string = "Madam, I'm Adam.";
33: 
34:     System.out.println();
35:     System.out.println("Testing whether the following " + "string is a palindrome:");
36:     System.out.println("    " + string);
37:     System.out.println();
38: 
39:     if (isPalindrome(string)) {
40:       System.out.println("It IS a palindrome!");
