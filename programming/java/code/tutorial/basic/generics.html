 5: 
 6: public class CountVowels {
 7: 
 8:   public static void main(String[] args) {
 9: 
10:     if (args.length < 1) {
11:       System.err.println("Usage: java CountVowels <word1 word2 ...>");
12:       System.exit(1);
13:     }
14: 
15:     Map<String, Integer> msi = new HashMap<>();
16: 
17:     for (String s : args) {
18:       int vowelCount = 0;
19:       int length = s.length();
20:       for (int i = 0; i < length; ++i) if (isVowel(s.charAt(i))) ++vowelCount;
21: 
22:       msi.put(s, vowelCount);
23:     }
24: 
25:     Set<String> words = msi.keySet();
26:     for (String wd : words) System.out.println(wd + ": " + msi.get(wd));
27:   }
28: 
29:   public static boolean isVowel(char c) {
30:     return c == 'a' || c == 'A' || c == 'e' || c == 'E' || c == 'i' || c == 'I' || c == 'o'
31:         || c == 'O' || c == 'u' || c == 'U';
32:   }
33: }

generics/MultiplicationTable.java

 1: import java.util.ArrayList;
 2: import java.util.Iterator;
 3: import java.util.List;
 4: import java.util.Set;
 5: import java.util.SortedMap;
 6: import java.util.TreeMap;
 7: 
 8: public class MultiplicationTable {
 9: 
10:   public static void main(String[] args) throws Exception {
11: 
12:     int begin = 1, end = 9;
13: 
14:     if (args.length == 2) {
15:       begin = Integer.parseInt(args[0]);
16:       end = Integer.parseInt(args[1]);
17:     }
18: 
19:     SortedMap<Integer, List<Integer>> mt = multTable(begin, end);
20:     printMultiplicationTable(mt);
21:     System.out.println();
22:   }
23: 
24:   public static SortedMap<Integer, List<Integer>> multTable(int begin, int end) {
25: 
26:     if (begin < 0 || end < 0 || begin > end) return null;
27: 
28:     SortedMap<Integer, List<Integer>> mt = new TreeMap<>();
29: 
30:     for (int i = begin; i <= end; ++i) {
31:       List<Integer> li = new ArrayList<>();
32:       for (int k = begin; k <= end; ++k) li.add(i * k);
33: 
34:       mt.put(i, li);
35:     }
36: 
37:     return mt;
38:   }
39: 
40:   public static void printMultiplicationTable(SortedMap<Integer, List<Integer>> mt) {
41: 
42:     Set<Integer> keys = mt.keySet();
43:     Integer[] iKeys = keys.toArray(new Integer[0]);
44:     Integer last = iKeys[iKeys.length - 1];
45:     int width = Integer.toString(last * last).length() + 1;
46:     String fmt = "%" + width + "d";
47: 
48:     System.out.printf("%" + width + "s", "");
49:     for (Integer k : keys) System.out.printf(fmt, k);
50:     System.out.println();
51: 
52:     for (Integer k : keys) {
53:       System.out.printf(fmt, k);
54: 
55:       List<Integer> li = mt.get(k);
56:       for (Integer i : li) System.out.printf(fmt, i);
57:       System.out.println();
58:     }
59:   }
60: }
