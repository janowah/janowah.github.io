 1: import java.util.ArrayList;
 2: import java.util.Collections;
 3: import java.util.List;
 4: 
 5: public class Deal {
 6:   public static void main(String[] args) {
 7:     if (args.length < 2) {
 8:       System.out.println("Usage: Deal hands cards");
 9:       return;
10:     }
11:     int numHands = Integer.parseInt(args[0]);
12:     int cardsPerHand = Integer.parseInt(args[1]);
13: 
14:     // Make a normal 52-card deck.
15:     String[] suit = new String[] {"spades", "hearts", "diamonds", "clubs"};
16:     String[] rank =
17:         new String[] {"ace", "2", "3", "4", "5", "6", "7", "8", "9", "10", "jack", "queen", "king"};
18:     List<String> deck = new ArrayList<String>();
19:     for (int i = 0; i < suit.length; i++) {
20:       for (int j = 0; j < rank.length; j++) {
21:         deck.add(rank[j] + " of " + suit[i]);
22:       }
23:     }
24: 
25:     // Shuffle the deck.
26:     Collections.shuffle(deck);
27: 
28:     if (numHands * cardsPerHand > deck.size()) {
29:       System.out.println("Not enough cards.");
30:       return;
31:     }
32: 
33:     for (int i = 0; i < numHands; i++) {
34:       System.out.println(dealHand(deck, cardsPerHand));
35:     }
36:   }
37: 
38:   public static <E> List<E> dealHand(List<E> deck, int n) {
39:     int deckSize = deck.size();
40:     List<E> handView = deck.subList(deckSize - n, deckSize);
41:     List<E> hand = new ArrayList<E>(handView);
42:     handView.clear();
43:     return hand;
44:   }
45: }

interfaces/Anagrams.java

 1: import java.io.File;
 2: import java.io.IOException;
 3: import java.util.ArrayList;
 4: import java.util.Arrays;
 5: import java.util.HashMap;
 6: import java.util.List;
 7: import java.util.Map;
 8: import java.util.Scanner;
 9: 
10: public class Anagrams {
11:   public static void main(String[] args) {
12:     int minGroupSize = Integer.parseInt(args[1]);
13: 
14:     // Read words from file and put into a simulated multimap
15:     Map<String, List<String>> m = new HashMap<String, List<String>>();
16: 
17:     try {
18:       Scanner s = new Scanner(new File(args[0]));
19:       while (s.hasNext()) {
20:         String word = s.next();
21:         String alpha = alphabetize(word);
22:         List<String> l = m.get(alpha);
23:         if (l == null) {
24:           m.put(alpha, l = new ArrayList<String>());
25:         }
26:         l.add(word);
27:       }
28:     } catch (IOException e) {
29:       System.err.println(e);
30:       System.exit(1);
31:     }
32: 
33:     // Print all permutation groups above size threshold
34:     for (List<String> l : m.values()) {
35:       if (l.size() >= minGroupSize) {
36:         System.out.println(l.size() + ": " + l);
37:       }
38:     }
39:   }
40: 
41:   private static String alphabetize(String s) {
42:     char[] a = s.toCharArray();
43:     Arrays.sort(a);
44:     return new String(a);
45:   }
46: }

22:     if (!(o instanceof Name)) {
23:       return false;

26:     return n.firstName.equals(firstName) && n.lastName.equals(lastName);
