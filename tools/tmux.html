tmux

C-b c ; create new
C-b x ; quit
C-b : ; command prompt

C-z suspend.
% unsuspend.

* sessions
  - tmux new -s session_name
  - tmux list-sessions
  - tmux detach ; C-b d
  - tmux attach -t session_name
  - tmux switch -t session_name

* help
  - tmux list-keys
  - tmux list-commands
  - tmux info
  - tmux source-file ~/.tmux.conf ; reload current config

C-b ? ; help. q to quit

* windows
  - C-b % ; vertical split
  - C-b { ; swap windows. C-b } also
  - C-b o ; cycle 

* editing
  - C-b [ ; copy
  - C-b ] ; paste

  - enter copy mode
  - select and copy text using applications functions
  - paste
